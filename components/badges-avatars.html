React TypeScript Props

type BadgeVariant = 'primary' | 'secondary' | 'success' | 'error' | 'warning' | 'neutral';
type BadgeSize    = 'sm' | 'md' | 'lg';

interface BadgeProps {
  variant?: BadgeVariant;
  size?: BadgeSize;
  /** Renders outline style instead of filled */
  outline?: boolean;
  /** Prepends a colored dot */
  dot?: boolean;
  children: React.ReactNode;
}

type AvatarSize    = 'xs' | 'sm' | 'md' | 'lg' | 'xl';
type AvatarVariant = 'primary' | 'secondary' | 'neutral';

interface AvatarProps {
  /** 1–2 letter initials shown when no image */
  initials: string;
  size?: AvatarSize;
  variant?: AvatarVariant;
  /** Shows green online dot */
  online?: boolean;
  /** Image URL — hides initials when provided */
  src?: string;
  alt?: string;
}

interface AvatarGroupProps {
  avatars: AvatarProps[];
  /** Max avatars before showing +N overflow */
  max?: number;
}

interface SpinnerProps {
  size?: 'sm' | 'md' | 'lg';
  variant?: 'primary' | 'secondary';
}

interface DividerProps {
  orientation?: 'horizontal' | 'vertical';
  /** Optional center label — renders labeled divider */
  label?: string;
}
