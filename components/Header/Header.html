Variant 3 — Mobile Bottom Nav

Floating pill fixed to the bottom of the screen. Contains 4 actions: Menu (opens the nav drawer), Citas (book appointment), Tarifas (billing), and WhatsApp.

Tapping Menu slides up the full navigation drawer over the page content. The bar remains visible below the drawer at all times.

Future React props — Full Header system

// ── Top bar (desktop) ─────────────────────────────────────────
interface DesktopHeaderProps {
  activeItem?: "atencion" | "especialidades" | "profesionales"
              | "investigacion" | "conoce";
  logoClinicaSrc?:   string;
  logoFundacionSrc?: string;
}

// ── Mobile top bar ────────────────────────────────────────────
interface MobileTopBarProps {
  logoClinicaSrc?:   string;
  logoFundacionSrc?: string;
}

// ── Mobile bottom nav ─────────────────────────────────────────
interface MobileBottomNavProps {
  activeNavItem?: "atencion" | "especialidades" | "profesionales"
                 | "investigacion" | "conoce";
  citasHref?:    string;   // defaults to "/agendar"
  tarifasHref?:  string;   // defaults to "/tarifas"
  whatsappHref?: string;   // defaults to "https://wa.me/57..."
}

// ── Composed mobile header (top + bottom) ─────────────────────
interface MobileHeaderProps extends MobileTopBarProps, MobileBottomNavProps {}

// ── Page-level usage ──────────────────────────────────────────
// Desktop: <DesktopHeader activeItem="atencion" />
// Mobile:  <MobileHeader activeNavItem="atencion" />
//   → renders MobileTopBar + MobileBottomNav as a unit
