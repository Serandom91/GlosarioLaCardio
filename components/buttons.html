React TypeScript Props

interface ButtonProps {
  /** Visual style variant */
  variant?: 'primary' | 'secondary' | 'ghost' | 'danger' | 'danger-outline';
  /** Size token */
  size?: 'lg' | 'md' | 'sm';
  /** Disabled state — renders muted grey, no pointer events */
  disabled?: boolean;
  /** Renders as a circle icon-only button (no text) */
  iconOnly?: boolean;
  /** Icon placed before label text */
  leftIcon?: React.ReactNode;
  /** Icon placed after label text */
  rightIcon?: React.ReactNode;
  /** Click handler */
  onClick?: (e: React.MouseEvent<HTMLButtonElement>) => void;
  /** Text content */
  children: React.ReactNode;
  /** When provided, renders an <a> tag instead of <button> */
  href?: string;
  /** HTML button type */
  type?: 'button' | 'submit' | 'reset';
  /** Extra classes */
  className?: string;
}
