React TypeScript Props

interface CheckboxProps {
  checked?: boolean;
  /** Three-state: shows dash instead of check */
  indeterminate?: boolean;
  disabled?: boolean;
  label?: string;
  /** Secondary description below label */
  description?: string;
  onChange?: (checked: boolean) => void;
}

interface RadioProps {
  name: string;
  value: string;
  checked?: boolean;
  disabled?: boolean;
  label: string;
  onChange?: (value: string) => void;
}

interface RadioGroupProps {
  name: string;
  value?: string;
  options: Array<{
    value: string;
    label: string;
    disabled?: boolean;
  }>;
  onChange?: (value: string) => void;
}

interface ToggleProps {
  checked?: boolean;
  disabled?: boolean;
  label?: string;
  onChange?: (checked: boolean) => void;
}
