React TypeScript Props

type TooltipPosition = 'top' | 'bottom' | 'left' | 'right';
type TooltipVariant  = 'dark' | 'light';

interface TooltipProps {
  /** Tooltip text or rich content */
  content: React.ReactNode;
  /** Preferred placement — auto-flips if viewport clips */
  position?: TooltipPosition;
  /** dark = near-black bg, light = white bg with shadow */
  variant?: TooltipVariant;
  /** Max width in px before text wraps */
  maxWidth?: number;
  /** The element that triggers the tooltip on hover / focus */
  children: React.ReactNode;
  /** Controlled visibility override */
  open?: boolean;
  /** Delay before showing, in ms */
  showDelay?: number;
}
